/* 1. Lex program to count the number of vowel and consontants in a given string*/

%{

	int vow_count = 0;
	int const_count = 0;

%}

%%
[aeiouAEIOU] {vow_count++;}
[a-zA-Z] {const_count++;}
%%

int yywrap() {}

int main() {
	printf("Enter thenumber of vowels & consonants");
	yylex();
	printf("No of vowels = %d\n No of consonants = %d", vow_count, const_count);

	return 0;

}
